58 void PermGroup::disjoint_decomp_generate_dependency_classes(
59  OrbitPartition &orbits) const
60 {
61  unsigned num_dependency_classes = 0u;
62 
63  std::vector<int> processed(orbits.num_partitions(), 0);
64  unsigned num_processed = 0u;
65 
66  for (unsigned i = 0u; i < orbits.num_partitions(); ++i) {
67  if (processed[i])
68  continue;
69 
70  // determine which orbits to merge
71  std::unordered_set<unsigned> merge{i};
72 
73  for (unsigned j = i + 1u; j < orbits.num_partitions(); ++j) {
74  if (processed[j])
75  continue;
76 
77  if (disjoint_decomp_orbits_dependent(orbits[i], orbits[j])) {
78  merge.insert(j);
79 
80  processed[j] = 1;
81  ++num_processed;
82  }
83  }
84 
85  // merge orbits
86  for (unsigned x = 1u; x <= degree(); ++x) {
87  if (merge.find(orbits.partition_index(x)) != merge.end())
88  orbits.change_partition(x, num_dependency_classes);
89  }
90 
91  ++num_dependency_classes;
92 
93  // check if we're done
94  processed[i] = 1;
95 
96  if (++num_processed == orbits.num_partitions())
97  break;
98  }

152 std::vector<PermGroup> PermGroup::disjoint_decomp_complete_recursive(
153  OrbitPartition const &orbits,
154  PermGroup const &perm_group)
155 {
156  // iterate over all possible partitions of the set of all orbits into two sets
157  assert(orbits.num_partitions() < 8 * sizeof(unsigned long long));
158 
159  for (auto part = 1ULL; !(part & (1ULL << (orbits.num_partitions() - 1u))); ++part) {
160  OrbitPartition orbit_split(perm_group.degree());
161 
162  for (unsigned x = 1u; x <= perm_group.degree(); ++x) {
163  if (orbits.partition_index(x) == -1)
164  continue;
165 
166  if ((1ULL << orbits.partition_index(x)) & part)
167  orbit_split.change_partition(x, 1);
168  else
169  orbit_split.change_partition(x, 0);
170  }
171 
172  DBG(TRACE) << "Considering orbit split:";
173  DBG(TRACE) << orbit_split;
174 
175  // try to find restricted subgroup decomposition
176  std::pair<PermGroup, PermGroup> restricted_subgroups;
177 
178  if (!disjoint_decomp_restricted_subgroups(
179  orbit_split, perm_group, restricted_subgroups))
180  continue;
181 
182  DBG(TRACE) << "Restricted groups are a disjoint subgroup decomposition";
183 
184  // recurse for both orbit partition elements and return combined result
185  auto orbits_recurse(orbits.split(orbit_split));
186 
187  DBG(TRACE) << "Recursing with orbit partitions:";
188  DBG(TRACE) << orbits_recurse[0];
189  DBG(TRACE) << orbits_recurse[1];
190 
191  return disjoint_decomp_join_results(
192  disjoint_decomp_complete_recursive(orbits_recurse[0],
193  restricted_subgroups.first),
194  disjoint_decomp_complete_recursive(orbits_recurse[1],
195  restricted_subgroups.second));
196  }
197 
198  DBG(TRACE) << "No further decomposition possible, returning group";
199 
200  return {perm_group};
201 }
