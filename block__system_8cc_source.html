55 PermSet BlockSystem::block_stabilizers(PermSet const &generators,
56  Block const &block)
57 {
58  // initialize block stabilizer generating set as generators of subgroup
59  // stabilizing a block element (we arbitrarily choose the first one)
60  PermGroup pg(generators.degree(), generators);
61 
62  pg.bsgs().base_change({block[0]});
63 
64  auto stabilizer_generators(pg.bsgs().stabilizers(0));
65 
66  auto stabilizer_orbit(Orbit::generate(block[0],
67  stabilizer_generators.with_inverses()));

70  std::unordered_set<unsigned> block_elements(block.begin(), block.end());
71 
72  for (unsigned beta : pg.bsgs().orbit(0)) {
73  if (block_elements.find(beta) == block_elements.end())
74  continue;

76  if (stabilizer_orbit.contains(beta))
77  continue;

88 BlockSystem BlockSystem::minimal(PermSet const &generators,
89  std::vector<unsigned> const &initial_block)
90 {
91  assert(initial_block.size() >= 2u);
92 
93  std::vector<unsigned> classpath(generators.degree());
94  std::vector<unsigned> cardinalities(generators.degree());
95  std::vector<unsigned> queue;
96 
97  DBG(DEBUG) << "Finding minimal block system for:";
98  DBG(DEBUG) << generators;
99 
100  DBG(TRACE) << "Initial block: " << initial_block;
101 
102  for (auto i = 0u; i < generators.degree(); ++i) {
103  classpath[i] = i;
104  cardinalities[i] = 1u;
105  }
106 
107  for (auto i = 0u; i < initial_block.size() - 1u; ++i) {
108  unsigned tmp = initial_block[i + 1u];
109 
110  classpath[tmp] = initial_block[0];
111  queue.push_back(tmp);
112  }
113 
114  cardinalities[initial_block[0]] = static_cast<unsigned>(initial_block.size());
115 
116  DBG(TRACE) << "Initial classpath: " << classpath;
117  DBG(TRACE) << "Initial cardinalities: " << cardinalities;
118  DBG(TRACE) << "Initial queue: " << queue;
119 
120  unsigned i = 0u;
121  unsigned l = initial_block.size() - 2u;
122 
123  while (i <= l) {
124  unsigned gamma = queue[i++];
125  DBG(TRACE) << "Gamma: " << gamma;
126 
127  for (Perm const &gen : generators) {
128  DBG(TRACE) << "Gen: " << gen;
129 
130  unsigned c1 = gen[gamma + 1u] - 1u;
131  unsigned c2 = gen[minimal_find_rep(gamma, classpath) + 1u] - 1u;
132 
133  DBG(TRACE) << "Considering classes " << c1 << " and " << c2;
134 
135  if (minimal_merge_classes(c1, c2, classpath, cardinalities, queue))
136  ++l;
137  }
138  }
139 
140  for (auto i = 0u; i < generators.degree(); ++i)
141  minimal_find_rep(i, classpath);
142 
143  minimal_compress_classpath(classpath);
144 
145  DBG(TRACE) << "Final classpath is: " << classpath;
146 
147  BlockSystem res(classpath);
148 
149  DBG(DEBUG) << "=> Resulting minimal block system:";
150  DBG(DEBUG) << res;
151 
152  return res;
153 }

243 {
244  auto maps_to_other_block = [&](Perm const &perm, unsigned x) {
245  return std::find(block.begin(), block.end(), perm[x]) == block.end();
246  };
247 
248  for (Perm const &gen : generators) {
249  bool should_map_to_other_block = maps_to_other_block(gen, block[0]);
250 
251  for (auto i = 1u; i < block.size(); ++i) {
252  if (maps_to_other_block(gen, block[i]) != should_map_to_other_block)
253  return false;
254  }

261  Block const &block)
262 {
263  assert(is_block(generators, block));
264 
265  std::vector<Block> blocks{block};
266 
267  std::vector<int> block_indices(generators.degree(), -1);
268  for (unsigned x : block)
269  block_indices[x - 1u] = 0;
270 
271  unsigned current_block_idx = 0u;
272  unsigned processed = block.size();
273 
274  while (current_block_idx < blocks.size()) {
275  auto current_block(blocks[current_block_idx]);
276 
277  for (Perm const &gen : generators) {
278  if (block_indices[gen[current_block[0]] - 1u] != -1)
279  continue;
280 
281  blocks.push_back(current_block.permuted(gen));
282 
283  for (unsigned x : current_block)
284  block_indices[gen[x] - 1u] = blocks.size();
285 
286  if ((processed += block.size()) == generators.degree())
287  return BlockSystem(blocks.begin(), blocks.end());
288  }

296 unsigned BlockSystem::minimal_find_rep(unsigned k,
297  std::vector<unsigned> &classpath)
298 {
299  // find class
300  unsigned res = k;
301  unsigned next = classpath[res];
302 
303  while (next != res) {
304  res = next;
305  next = classpath[res];
306  }
307 
308  // compress path
309  unsigned current = k;
310  next = classpath[k];
311 
312  while (next != current) {
313  classpath[current] = res;
314 
315  current = next;
316  next = classpath[current];
317  }

319  return res;

406 std::vector<BlockSystem> BlockSystem::non_trivial_non_transitive(
407  PermGroup const &pg)
408 {
409  OrbitPartition orbits(pg.degree(), pg.generators());
410 
411  DBG(TRACE) << "Group has " << orbits.num_partitions() << " distinct orbits:";
412 #ifndef NDEBUG
413  for (auto const &orbit : orbits)
414  DBG(TRACE) << orbit;
415 #endif
416 
417  std::vector<std::vector<BlockSystem>> partial_blocksystems(orbits.num_partitions());
418  std::vector<unsigned> domain_offsets(orbits.num_partitions());
419 
420  for (auto i = 0u; i < orbits.num_partitions(); ++i) {
421  // calculate all non trivial block systems for orbit restricted group
422  PermSet restricted_gens;
423 
424  auto orbit_extremes =
425  std::minmax_element(orbits[i].begin(), orbits[i].end());
426 
427  unsigned orbit_low = *std::get<0>(orbit_extremes);
428  unsigned orbit_high = *std::get<1>(orbit_extremes);
429 
430  domain_offsets[i] = orbit_low - 1u;
431 
432  for (Perm const &gen : pg.generators()) {
433  Perm perm(gen.restricted(orbits[i].begin(), orbits[i].end()));
434 
435  if (!perm.id())
436  restricted_gens.insert(perm.normalized(orbit_low, orbit_high));
437  }
438 
439  DBG(TRACE) << "Group generators restricted to " << orbits[i] << ":";
440  DBG(TRACE) << restricted_gens;
441 
442  PermGroup pg_restricted(orbit_high - orbit_low + 1u, restricted_gens);
443  partial_blocksystems[i] = non_trivial(pg_restricted, true);
444 
445  // append trivial blocksystem {{x} | x in orbit}
446  std::vector<unsigned> trivial_classes(orbits[i].size());
447  std::iota(trivial_classes.begin(), trivial_classes.end(), 0u);
448 
449  BlockSystem bs(trivial_classes);
450  partial_blocksystems[i].push_back(bs);
451  }
452 
453  DBG(TRACE) << "Relevant block systems for all group restrictions:";
454 #ifndef NDEBUG
455  for (auto const &bs : partial_blocksystems)
456  DBG(TRACE) << bs;
457 #endif
458 
459  auto blocksystem_representatives(
460  non_trivial_find_representatives(pg.generators(),
461  partial_blocksystems,
462  domain_offsets));
463 
464  return non_trivial_from_representatives(pg.generators(),
465  blocksystem_representatives);
466 }

468 std::vector<BlockSystem::Block> BlockSystem::non_trivial_find_representatives(
469  PermSet const &generators,
470  std::vector<std::vector<BlockSystem>> const &partial_blocksystems,
471  std::vector<unsigned> const &domain_offsets)
472 {
473  DBG(TRACE) << "Finding block system representatives";
474 
475  std::vector<Block> res;
476 
477  std::function<void(std::vector<BlockSystem const *> const &, unsigned, bool)>
478  recurse = [&](std::vector<BlockSystem const *> const &current_blocksystems,
479  unsigned i,
480  bool one_trivial)
481  {
482  if (i == partial_blocksystems.size()) {
483  DBG(TRACE) << "Considering block system combination:";
484 #ifndef NDEBUG
485  for (auto j = 0u; j < current_blocksystems.size(); ++j) {
486  DBG(TRACE) << *current_blocksystems[j]
487  << " (shifted by " << domain_offsets[j] << ")";
488  }
489 #endif
490 
491  Block current_block_unshifted((*current_blocksystems[0])[0]);
492  Block current_block(current_block_unshifted.shifted(domain_offsets[0]));
493 
494  for (auto j = 1u; j < current_blocksystems.size(); ++j) {
495  bool next_block = false;
496 
497  for (auto const &block : *current_blocksystems[j]) {
498  Block shifted_block(block.shifted(domain_offsets[j]));
499  Block extended_block(current_block.unified(shifted_block));
500 
501  if (is_block(generators, extended_block)) {
502  current_block = extended_block;
503  next_block = true;
504  break;
505  }
506  }
507 
508  if (!next_block)
509  return;
510  }
511 
512  res.push_back(current_block);
513  DBG(TRACE) << "Found representative block: " << current_block;
514 
515  return;
516  }
517 
518  for (BlockSystem const &blocksystem : partial_blocksystems[i]) {
519  if (blocksystem.trivial()) {
520  if (one_trivial)
521  return;
522 
523  one_trivial = true;
524  }
525 
526  auto extended_blocksystems(current_blocksystems);
527  extended_blocksystems.push_back(&blocksystem);
528 
529  recurse(extended_blocksystems, i + 1, one_trivial);
530  }
531  };
532 
533  recurse({}, 0u, false);
534 
535  return res;
536 }
